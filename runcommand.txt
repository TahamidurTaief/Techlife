for first time: python manage.py tailwind setup


Terminal 1: python manage.py runserver
Terminal 2: python manage.py tailwind watch

or : python manage.py tailwind runserver
  



# install uv
pip install uv

# init uv
uv init


# install venv
uv venv

# active venv
source .venv/Scripts/activate



# install packages
uv add -r requirements.txt


# Build Tailwind CSS for production
uv run manage.py tailwind build


# Run development server with Tailwind watch
uv run manage.py tailwind runserver


# Create new Django app
uv run manage.py startapp app_name

# Make migrations
python manage.py makemigrations

# Apply migrations
python manage.py migrate
uv venv

# Create superuser
python manage.py createsuperuser

# Collect static files
python manage.py collectstatic
